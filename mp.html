        #playPauseBtn, #forwardBtn, #backwardBtn {

            font-size: 20px;

            width: 80%;

<body>

    <div class="player-container">

        <video id="videoPlayer" width="1920" height="1080" controls>

            <source src=“input_video.mp4" type="video/mp4">

            Your browser does not support the video tag.

        </video>

        <div class="controls">

            <button id="backwardBtn">&lt;&lt;</button>

            <button id="playPauseBtn">&#9658;</button>

            <button id="forwardBtn">&gt;&gt;</button>

            <input type="range" id="volumeControl" min="0" max="1" step="0.1" value="1">

            <input type="range" id="progressBar" value="0" max="100">

        </div>

    </div>

    <script>

        document.addEventListener("DOMContentLoaded", function () {

            const videoPlayer = document.getElementById("videoPlayer");

            const playPauseBtn = document.getElementById("playPauseBtn");

            const volumeControl = document.getElementById("volumeControl");

            const progressBar = document.getElementById("progressBar");

            const forwardBtn = document.getElementById("forwardBtn");

            const backwardBtn = document.getElementById("backwardBtn");


            playPauseBtn.addEventListener("click", togglePlayPause);

            volumeControl.addEventListener("input", adjustVolume);

            videoPlayer.addEventListener("timeupdate", updateProgressBar);

            progressBar.addEventListener("input", seek);

            forwardBtn.addEventListener("click", forward);

            backwardBtn.addEventListener("click", backward);


            function togglePlayPause() {

                if (videoPlayer.paused) {

                    videoPlayer.play();

                    playPauseBtn.innerHTML = "&#10074;&#10074;"; // Pause symbol

                } else {

                    videoPlayer.pause();

                    playPauseBtn.innerHTML = "&#9658;"; // Play symbol

                }

            }


            function adjustVolume() {

                videoPlayer.volume = volumeControl.value;

            }


            function updateProgressBar() {

                const percentage = (videoPlayer.currentTime / videoPlayer.duration) * 100;

                progressBar.value = percentage;

            }


            function seek() {

                const newPosition = (progressBar.value / 100) * videoPlayer.duration;

                videoPlayer.currentTime = newPosition;

            }


            function forward() {

                videoPlayer.currentTime += 5; // Forward by 5 seconds

            }


            function backward() {

                videoPlayer.currentTime -= 5; // Backward by 5 seconds

            }

        });

    </script>

</body>
